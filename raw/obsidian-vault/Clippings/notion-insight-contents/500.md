def _call_llm(self, prompt: str, temperature: float = 0.7) -> str:
    """Ollama API 호출"""
    try:
        response = requests.post(
            f"{self.base_url}/api/generate",
            json={
                "model": self.model,
                "prompt": prompt,
                "stream": False,
                "options": {
                    "temperature": temperature,
                    "num_predict": 4000,
                }
            },
            timeout=600  # 10분 타임아웃 (70B 모델은 느림)
        )
        response.raise_for_status()
        return response.json().get("response", "")
    except Exception as e:
        raise Exception(f"LLM 호출 실패: {str(e)}")

def analyze_market_condition(self, df: pd.DataFrame, lookback_days: int = 30) -> Dict:
    """
    최근 시장 상황 분석 (nof1.ai 스타일)
    LLM에게 제공할 시장 컨텍스트 생성

    Args:
        df: OHLCV 데이터프레임 (timestamp, open, high, low, close, volume, bb_* 등)
        lookback_days: 분석할 최근 일수

    Returns:
        시장 상황 분석 결과 딕셔너리
    """
    if df.empty:
        return {"error": "데이터가 비어있습니다"}

    # 최근 데이터 필터링
    df = df.copy()
    if 'timestamp' in df.columns:
        df['timestamp'] = pd.to_datetime(df['timestamp'])
        cutoff = df['timestamp'].max() - timedelta(days=lookback_days)
        recent_df = df[df['timestamp'] >= cutoff].copy()
    else:
        # 마지막 N개 봉 사용 (30분봉 기준 하루 약 48개)
        bars_per_day = 48
        recent_df = df.tail(lookback_days * bars_per_day).copy()

    if recent_df.empty:
        return {"error": "최근 데이터가 없습니다"}

    # 1. 가격 동향 분석
    start_price = recent_df['close'].iloc[0]
    end_price = recent_df['close'].iloc[-1]
    price_change = end_price - start_price
    price_change_pct = (price_change / start_price) * 100
    high_price = recent_df['high'].max()
    low_price = recent_df['low'].min()
    price_range = high_price - low_price

    # 2. 추세 분석
    closes = recent_df['close'].values
    x = np.arange(len(closes))
    if len(closes) > 1:
        slope, intercept = np.polyfit(x, closes, 1)
        trend_direction = "상승" if slope > 0.1 else ("하락" if slope < -0.1 else "횡보")
        trend_strength = abs(slope) * len(closes)  # 기간 내 예상 변화량
    else:
        trend_direction = "판단불가"
        trend_strength = 0

    # 3. 변동성 분석
    if 'atr_20' in recent_df.columns and recent_df['atr_20'].notna().any():
        avg_atr = recent_df['atr_20'].mean()
        current_atr = recent_df['atr_20'].iloc[-1]
    else:
        # ATR 직접 계산
        recent_df['tr'] = np.maximum(
            recent_df['high'] - recent_df['low'],
            np.maximum(
                abs(recent_df['high'] - recent_df['close'].shift(1)),
                abs(recent_df['low'] - recent_df['close'].shift(1))
            )
        )
        avg_atr = recent_df['tr'].rolling(20).mean().mean()
        current_atr = recent_df['tr'].rolling(20).mean().iloc[-1] if len(recent_df) >= 20 else recent_df['tr'].mean()

    volatility_level = "높음" if current_atr > avg_atr * 1.2 else ("낮음" if current_atr < avg_atr * 0.8 else "보통")

    # 4. BB 밴드폭 분석
    if 'bb_width' in recent_df.columns and recent_df['bb_width'].notna().any():
        avg_bb_width = recent_df['bb_width'].mean()
        current_bb_width = recent_df['bb_width'].iloc[-1]
    elif all(col in recent_df.columns for col in ['bb_upper', 'bb_lower', 'bb_mid']):
        recent_df['bb_width'] = (recent_df['bb_upper'] - recent_df['bb_lower']) / recent_df['bb_mid'] * 100
        avg_bb_width = recent_df['bb_width'].mean()
        current_bb_width = recent_df['bb_width'].iloc[-1]
    else:
        avg_bb_width = 0
        current_bb_width = 0

    bb_state = "확장" if current_bb_width > avg_bb_width * 1.2 else ("수축" if current_bb_width < avg_bb_width * 0.8 else "보통")

    # 5. BB 이탈 횟수 분석
    if all(col in recent_df.columns for col in ['close', 'bb_lower', 'bb_upper']):
        lower_breaks = (recent_df['close'] < recent_df['bb_lower']).sum()
        upper_breaks = (recent_df['close'] > recent_df['bb_upper']).sum()
    else:
        lower_breaks = 0
        upper_breaks = 0

    # 6. 거래량 분석
    if 'volume' in recent_df.columns and recent_df['volume'].notna().any():
        avg_volume = recent_df['volume'].mean()
        recent_volume = recent_df['volume'].tail(10).mean()
        volume_trend = "증가" if recent_volume > avg_volume * 1.2 else ("감소" if recent_volume < avg_volume * 0.8 else "보통")
    else:
        avg_volume = 0
        recent_volume = 0
        volume_trend = "데이터없음"

    # 7. 일별 수익률 분포 (최근 20일)
    if 'timestamp' in recent_df.columns:
        daily_returns = recent_df.groupby(recent_df['timestamp'].dt.date)['close'].last().pct_change().dropna()
        if len(daily_returns) > 0:
            win_days = (daily_returns > 0).sum()
            lose_days = (daily_returns < 0).sum()
            avg_daily_return = daily_returns.mean() * 100
            max_daily_gain = daily_returns.max() * 100
            max_daily_loss = daily_returns.min() * 100
        else:
            win_days = lose_days = 0
            avg_daily_return = max_daily_gain = max_daily_loss = 0
    else:
        win_days = lose_days = 0
        avg_daily_return = max_daily_gain = max_daily_loss = 0

    # 8. RSI 분석 (있으면)
    if 'rsi' in recent_df.columns and recent_df['rsi'].notna().any():
        current_rsi = recent_df['rsi'].iloc[-1]
        rsi_state = "과매수" if current_rsi > 70 else ("과매도" if current_rsi < 30 else "중립")
    else:
        current_rsi = None
        rsi_state = "데이터없음"

    # 결과 정리
    market_analysis = {
        "period": {
            "lookback_days": lookback_days,
            "bars_analyzed": len(recent_df),
            "start_date": str(recent_df['timestamp'].iloc[0])[:10] if 'timestamp' in recent_df.columns else "N/A",
            "end_date": str(recent_df['timestamp'].iloc[-1])[:10] if 'timestamp' in recent_df.columns else "N/A",
        },
        "price": {
            "start": round(start_price, 2),
            "end": round(end_price, 2),
            "change": round(price_change, 2),
            "change_pct": round(price_change_pct, 2),
            "high": round(high_price, 2),
            "low": round(low_price, 2),
            "range": round(price_range, 2),
        },
        "trend": {
            "direction": trend_direction,
            "strength": round(trend_strength, 2),
            "slope_per_bar": round(slope, 4) if len(closes) > 1 else 0,
        },
        "volatility": {
            "current_atr": round(current_atr, 2) if current_atr else 0,
            "avg_atr": round(avg_atr, 2) if avg_atr else 0,
            "level": volatility_level,
        },
        "bollinger_bands": {
            "current_width_pct": round(current_bb_width, 2),
            "avg_width_pct": round(avg_bb_width, 2),
            "state": bb_state,
            "lower_breaks": int(lower_breaks),
            "upper_breaks": int(upper_breaks),
        },
        "volume": {
            "avg": int(avg_volume) if avg_volume else 0,
            "recent": int(recent_volume) if recent_volume else 0,
            "trend": volume_trend,
        },
        "daily_returns": {
            "win_days": int(win_days),
            "lose_days": int(lose_days),
            "avg_return_pct": round(avg_daily_return, 2),
            "max_gain_pct": round(max_daily_gain, 2),
            "max_loss_pct": round(max_daily_loss, 2),
        },
        "rsi": {
            "current": round(current_rsi, 1) if current_rsi else None,
            "state": rsi_state,
        },
        # 종합 시장 상태
        "summary": {
            "market_regime": self._determine_market_regime(trend_direction, volatility_level, bb_state),
            "recommended_strategy_type": self._recommend_strategy_type(trend_direction, volatility_level, price_change_pct),
        }
    }

    return market_analysis

def _determine_market_regime(self, trend: str, volatility: str, bb_state: str) -> str:
    """시장 체제(regime) 판단"""
    if trend == "상승" and volatility == "낮음":
        return "안정적 상승장"
    elif trend == "상승" and volatility == "높음":
        return "변동성 상승장"
    elif trend == "하락" and volatility == "높음":
        return "급락장"
    elif trend == "하락" and volatility == "낮음":
        return "완만한 하락장"
    elif trend == "횡보" and bb_state == "수축":
        return "횡보 수렴장 (돌파 대기)"
    elif trend == "횡보" and bb_state == "확장":
        return "변동성 횡보장"
    else:
        return "혼조장"

def _recommend_strategy_type(self, trend: str, volatility: str, price_change_pct: float) -> str:
    """시장 상황에 맞는 전략 유형 추천"""
    recommendations = []

    if trend == "상승":
        recommendations.append("롱 위주 전략")
        if volatility == "높음":
            recommendations.append("넓은 손절 필요")
        else:
            recommendations.append("눌림목 매수")
    elif trend == "하락":
        recommendations.append("숏 위주 전략")
        if volatility == "높음":
            recommendations.append("반등 매도")
        else:
            recommendations.append("보수적 진입")
    else:  # 횡보
        recommendations.append("양방향 전략")
        recommendations.append("BB 이탈 역추세 유효")

    if abs(price_change_pct) > 5:
        recommendations.append("추세추종 유리")
    elif abs(price_change_pct) < 2:
        recommendations.append("평균회귀 유리")

    return " / ".join(recommendations)

## 과적합 문제 분석

Walk-Forward 검증 실패는 전략이 **과적합(overfitting)**되었음을 의미합니다.

- In-Sample(학습 기간)에서는 좋은 성과를 보였지만
- Out-of-Sample(미래 데이터)에서는 성과가 나빠짐
- 이는 전략이 과거 데이터의 노이즈까지 학습했기 때문

1. **파라미터 단순화**: 복잡한 필터와 조건을 줄이기
2. **손절폭 확대**: ATR 배수를 늘려 노이즈에 덜 민감하게 (예: 3.5, 4.0, 4.5 등)

4. **거래 빈도 줄이기**: 더 엄격한 진입 조건
5. **필터 제거**: 과최적화의 원인이 되는 필터 비활성화

def _find_best_result(self, results_history: List[Dict]) -> Dict:
    """
    최고 성과 전략 찾기 (프론트엔드와 동일한 로직)
    우선순위:
    0. 의심스러운 결과 제외 (suspicious=True)
    1. WF 통과 전략 중 최고 수익
    2. WF 검증된 전략 중 최고 WF 점수
    3. 총 수익 기준
    """
    if not results_history:
        return {}

    # 0. 의심스러운 결과 제외
    non_suspicious = [
        r for r in results_history
        if not r.get('metrics', {}).get('suspicious', False)
    ]

    # 모든 결과가 의심스러우면 원본 사용 (경고와 함께)
    candidates = non_suspicious if non_suspicious else results_history

    # 1. WF 통과 전략들 중에서 최고 수익 선택
    wf_passed_results = [
        r for r in candidates
        if r.get('wf_result', {}).get('passed', False)
    ]
    if wf_passed_results:
        return max(wf_passed_results,
                   key=lambda x: x.get('metrics', {}).get('total_pnl', float('-inf')))

    # 2. WF 통과 전략이 없으면 WF 검증된 것 중에서 선택
    # 점수가 비슷하면(10% 이내) 수익 기준으로 선택
    wf_tested_results = [r for r in candidates if r.get('wf_result')]
    if wf_tested_results:
        def compare_wf_results(a, b):
            score_a = a.get('wf_result', {}).get('holdout_generalization_score', 0)
            score_b = b.get('wf_result', {}).get('holdout_generalization_score', 0)
            pnl_a = a.get('metrics', {}).get('total_pnl', 0)
            pnl_b = b.get('metrics', {}).get('total_pnl', 0)

            # 점수 차이가 10% 이내면 수익 기준
            score_diff = abs(score_a - score_b)
            max_score = max(score_a, score_b)
            is_similar = (score_diff / max_score < 0.1) if max_score > 0 else True

            if is_similar:
                return a if pnl_a >= pnl_b else b
            return a if score_a >= score_b else b

        best = wf_tested_results[0]
        for r in wf_tested_results[1:]:
            best = compare_wf_results(best, r)
        return best

    # 3. WF 검증 없으면 총 수익 기준 (의심스럽지 않은 결과 우선)
    return max(candidates,
               key=lambda x: x.get('metrics', {}).get('total_pnl', float('-inf')))
